## 二、Docker 容器化部署

### 2.1 目录结构

```
ft-deploy/
├── docker/
│   ├── docker-compose.yml          # 主编排文件
│   ├── docker-compose.yml         # 生产环境
│   ├── mysql/
│   │   └── my.cnf                 # MySQL 配置
│   ├── redis/
│   │   └── redis.conf             # Redis 配置
│   ├── nginx/
│   │   ├── nginx.conf             # Nginx 主配置
│   │   └── conf.d/
│   │       └── ft.conf            # FT 项目配置
│   ├── minio/
│   │   └── data/                  # MinIO 数据目录
│   └── es/
│       └── elasticsearch.yml      # ES 配置
├── backend/
│   ├── Dockerfile                 # 后端镜像构建
│   └── wait-for-it.sh            # 等待脚本
├── frontend/
│   ├── Dockerfile                # 前端镜像构建
│   └── nginx.conf                # 前端 Nginx 配置
└── scripts/
    ├── init-db.sh                # 数据库初始化
    └── backup.sh                 # 备份脚本
```

### 2.2 docker-compose.yml

```yaml
version: '3.8'

services:
  # MySQL 数据库
  mysql:
    image: mysql:8.0
    container_name: ft-mysql
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root_password}
      MYSQL_DATABASE: srs_db
      TZ: Asia/Shanghai
    ports:
      - "3306:3306"
    volumes:
      - ./mysql/data:/var/lib/mysql
      - ./mysql/my.cnf:/etc/mysql/conf.d/my.cnf
      - ./mysql/init:/docker-entrypoint-initdb.d
    command: --default-authentication-plugin=mysql_native_password
    networks:
      - ft-network
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: ft-redis
    restart: always
    ports:
      - "6379:6379"
    volumes:
      - ./redis/data:/data
      - ./redis/redis.conf:/etc/redis/redis.conf
    command: redis-server /etc/redis/redis.conf
    networks:
      - ft-network
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3

  # Nacos 注册中心
  nacos:
    image: nacos/nacos-server:v2.2.0
    container_name: ft-nacos
    restart: always
    environment:
      MODE: standalone
      SPRING_DATASOURCE_PLATFORM: mysql
      MYSQL_SERVICE_HOST: mysql
      MYSQL_SERVICE_DB_NAME: srs_db
      MYSQL_SERVICE_PORT: 3306
      MYSQL_SERVICE_USER: root
      MYSQL_SERVICE_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root_password}
      MYSQL_SERVICE_DB_PARAM: characterEncoding=utf8&connectTimeout=1000&socketTimeout=3000&autoReconnect=true&useSSL=false&serverTimezone=Asia/Shanghai
      JVM_XMS: 512m
      JVM_XMX: 512m
      JVM_XMN: 256m
    ports:
      - "8848:8848"
      - "9848:9848"
    volumes:
      - ./nacos/logs:/home/nacos/logs
      - ./nacos/data:/home/nacos/data
    depends_on:
      mysql:
        condition: service_healthy
    networks:
      - ft-network

  # RocketMQ NameServer
  rocketmq-namesrv:
    image: apache/rocketmq:4.9.4
    container_name: ft-rocketmq-namesrv
    restart: always
    environment:
      JAVA_OPT: -server -Xms256m -Xmx256m -Xmn128m
    ports:
      - "9876:9876"
    volumes:
      - ./rocketmq/namesrv/logs:/root/logs
      - ./rocketmq/namesrv/store:/root/store
    command: sh mqnamesrv
    networks:
      - ft-network

  # RocketMQ Broker
  rocketmq-broker:
    image: apache/rocketmq:4.9.4
    container_name: ft-rocketmq-broker
    restart: always
    environment:
      JAVA_OPT: -server -Xms512m -Xmx512m -Xmn256m
    ports:
      - "10911:10911"
      - "10912:10912"
    volumes:
      - ./rocketmq/broker/logs:/root/logs
      - ./rocketmq/broker/store:/root/store
      - ./rocketmq/broker/conf/broker.conf:/opt/rocketmq/conf/broker.conf
    depends_on:
      - rocketmq-namesrv
    command: sh mqbroker -c /opt/rocketmq/conf/broker.conf
    networks:
      - ft-network

  # Elasticsearch
  elasticsearch:
    image: elasticsearch:7.17.6
    container_name: ft-elasticsearch
    restart: always
    environment:
      - discovery.type=single-node
      - ES_JAVA_OPTS=-Xms1g -Xmx1g
      - xpack.security.enabled=false
    ports:
      - "9200:9200"
      - "9300:9300"
    volumes:
      - ./es/data:/usr/share/elasticsearch/data
      - ./es/logs:/usr/share/elasticsearch/logs
      - ./es/elasticsearch.yml:/usr/share/elasticsearch/config/elasticsearch.yml
    networks:
      - ft-network
    healthcheck:
      test: ["CMD-SHELL", "curl -f http://localhost:9200/_cluster/health || exit 1"]
      interval: 30s
      timeout: 10s
      retries: 5

  # MinIO 对象存储
  minio:
    image: minio/minio:latest
    container_name: ft-minio
    restart: always
    environment:
      MINIO_ROOT_USER: ${MINIO_ACCESS_KEY:-minioadmin}
      MINIO_ROOT_PASSWORD: ${MINIO_SECRET_KEY:-minioadmin}
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - ./minio/data:/data
    command: server /data --console-address ":9001"
    networks:
      - ft-network
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9000/minio/health/live"]
      interval: 30s
      timeout: 20s
      retries: 3

  # srs-center 统一认证中心
  srs-center:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: srs-center
    restart: always
    environment:
      SPRING_PROFILES_ACTIVE: prod
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/srs_db?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimezone=Asia/Shanghai
      SPRING_DATASOURCE_USERNAME: root
      SPRING_DATASOURCE_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root_password}
      SPRING_REDIS_HOST: redis
      SPRING_REDIS_PORT: 6379
      ROCKETMQ_NAMESRV_ADDR: rocketmq-namesrv:9876
      NACOS_SERVER_ADDR: nacos:8848
    ports:
      - "8081:8081"
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_healthy
    networks:
      - ft-network

  # srs-ctrl 智能管控平台
  srs-ctrl:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: srs-ctrl
    restart: always
    environment:
      SPRING_PROFILES_ACTIVE: prod
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/srs_db?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimezone=Asia/Shanghai
      SPRING_DATASOURCE_USERNAME: root
      SPRING_DATASOURCE_PASSWORD: ${MYSQL_ROOT_PASSWORD:-root_password}
      SPRING_REDIS_HOST: redis
      SPRING_REDIS_PORT: 6379
      ROCKETMQ_NAMESRV_ADDR: rocketmq-namesrv:9876
      NACOS_SERVER_ADDR: nacos:8848
    ports:
      - "8082:8082"
    depends_on:
      mysql:
        condition: service_healthy
      redis:
        condition: service_healthy
    networks:
      - ft-network

  # srs-gateway 安全网关
  srs-gateway:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: srs-gateway
    restart: always
    environment:
      SPRING_PROFILES_ACTIVE: prod
      NACOS_SERVER_ADDR: nacos:8848
    ports:
      - "8080:8080"
    depends_on:
      - nacos
    networks:
      - ft-network

  # Nginx 负载均衡
  nginx:
    image: nginx:alpine
    container_name: ft-nginx
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/conf.d:/etc/nginx/conf.d
      - ./frontend/dist:/usr/share/nginx/html
      - ./nginx/logs:/var/log/nginx
      - ./nginx/ssl:/etc/nginx/ssl
    depends_on:
      - srs-center
      - srs-ctrl
      - srs-gateway
    networks:
      - ft-network
    healthcheck:
      test: ["CMD", "nginx", "-t"]
      interval: 30s
      timeout: 10s
      retries: 3

networks:
  ft-network:
    driver: bridge
    ipam:
      config:
        - subnet: 172.20.0.0/16
```

### 2.3 后端 Dockerfile

```dockerfile
FROM openjdk:17-slim

# 安装必要工具
RUN apt-get update && apt-get install -y \
    curl \
    && rm -rf /var/lib/apt/lists/*

# 创建应用目录
RUN mkdir -p /app

# 复制构建产物
COPY ft-*/target/*.jar /app/app.jar

# 设置工作目录
WORKDIR /app

# JVM 参数优化
ENV JAVA_OPTS="-Xms512m -Xmx1024m -XX:+UseG1GC -XX:+HeapDumpOnOutOfMemoryError"

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8080/actuator/health || exit 1

# 启动命令
ENTRYPOINT ["sh", "-c", "java $JAVA_OPTS -jar /app/app.jar"]
```

### 2.4 前端 Dockerfile

```dockerfile
# 构建阶段
FROM node:18-alpine as builder

WORKDIR /app

# 复制 package.json
COPY package.json yarn.lock ./
RUN yarn install --frozen-lockfile

# 复制源代码
COPY . .

# 构建生产版本
RUN yarn build

# 运行阶段
FROM nginx:alpine

# 复制构建产物
COPY --from=builder /app/dist /usr/share/nginx/html

# 复制 Nginx 配置
COPY nginx.conf /etc/nginx/conf.d/default.conf

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"}
```

### 2.5 前端 nginx.conf

```nginx
server {
    listen 80;
    server_name localhost;
    root /usr/share/nginx/html;
    index index.html;

    # 启用 gzip
    gzip on;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml;
    gzip_min_length 1024;

    # 前端路由 (SPA)
    location / {
        try_files $uri $uri/ /index.html;

        # 缓存静态资源
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }

    # API 代理 - 统一认证中心
    location /api/center/ {
        proxy_pass http://srs-center:8081/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # API 代理 - 智能管控平台
    location /api/ctrl/ {
        proxy_pass http://srs-ctrl:8082/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # API 代理 - 网关
    location /api/gateway/ {
        proxy_pass http://srs-gateway:8080/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # 错误页面
    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
        root /usr/share/nginx/html;
    }
}
```

### 2.6 启动脚本

**deploy.sh:**

```bash
#!/bin/bash

# 星环空间部署脚本

set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

echo -e "${GREEN}=== 星环空间部署脚本 ===${NC}"

# 检查 Docker
if ! command -v docker &> /dev/null; then
    echo -e "${RED}错误: Docker 未安装${NC}"
    exit 1
fi

if ! command -v docker-compose &> /dev/null; then
    echo -e "${RED}错误: Docker Compose 未安装${NC}"
    exit 1
fi

# 创建网络
docker network create ft-network 2>/dev/null || true

# 启动基础设施服务 (MySQL, Redis, Nacos 等)
echo -e "${YELLOW}启动基础设施服务...${NC}"
docker-compose up -d mysql redis nacos

# 等待 MySQL 就绪
echo -e "${YELLOW}等待 MySQL 就绪...${NC}"
sleep 30

# 初始化数据库
echo -e "${YELLOW}初始化数据库...${NC}"
docker exec ft-mysql mysql -uroot -p${MYSQL_ROOT_PASSWORD:-root_password} < ../init_mysql.sql

# 启动中间件
echo -e "${YELLOW}启动中间件服务...${NC}"
docker-compose up -d rocketmq-namesrv rocketmq-broker elasticsearch minio

# 等待中间件就绪
echo -e "${YELLOW}等待中间件就绪...${NC}"
sleep 20

# 构建后端镜像
echo -e "${YELLOW}构建后端服务镜像...${NC}"
cd backend
mvn clean package -DskipTests
docker build -t ft-backend:latest .

# 构建前端镜像
echo -e "${YELLOW}构建前端服务镜像...${NC}"
cd ../frontend
yarn install
yarn build
docker build -t ft-frontend:latest .

# 启动应用服务
echo -e "${YELLOW}启动应用服务...${NC}"
cd ..
docker-compose up -d srs-center srs-ctrl srs-gateway nginx

echo -e "${GREEN}=== 部署完成 ===${NC}"
echo "前端地址: http://localhost"
echo "Nacos 控制台: http://localhost:8848/nacos (nacos/nacos)"
echo "MinIO 控制台: http://localhost:9001 (minioadmin/minioadmin)"
```

## 四、数据备份策略

### 4.1 备份脚本

**scripts/backup.sh:**

```bash
#!/bin/bash

BACKUP_DIR=/data/backup
DATE=$(date +%Y%m%d%H%M%S)

# 创建备份目录
mkdir -p ${BACKUP_DIR}

# 备份 MySQL
echo "备份 MySQL 数据库..."
docker exec ft-mysql mysqldump -uroot -p${MYSQL_ROOT_PASSWORD} srs_db | gzip > ${BACKUP_DIR}/mysql_${DATE}.sql.gz

# 备份 Redis
echo "备份 Redis 数据..."
docker exec ft-redis redis-cli SAVE
docker cp ft-redis:/data/dump.rdb ${BACKUP_DIR}/redis_${DATE}.rdb

# 备份 MinIO
echo "备份 MinIO 数据..."
mc mirror minio/xx-ft ${BACKUP_DIR}/minio_${DATE}/

# 备份 Elasticsearch
echo "备份 Elasticsearch 数据..."
curl -X POST "http://localhost:9200/_snapshot/ft_backup/snapshot_${DATE}?wait_for_completion=true"

# 清理 30 天前的备份
find ${BACKUP_DIR} -type f -mtime +30 -delete

echo "备份完成: ${DATE}"
```

### 4.2 定时任务

```bash
# 每天凌晨 2 点执行备份
0 2 * * * /opt/ft-deploy/scripts/backup.sh >> /var/log/backup.log 2>&1
```
