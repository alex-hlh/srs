## 二、密码安全

### 2.1 密码加密策略

```
┌────────────────────────────────────────────────────────────────────────────┐
│                          密码加密存储流程                                  │
└────────────────────────────────────────────────────────────────────────────┘

用户输入密码: "123456"
        │
        ▼
┌─────────────────┐
│  1.生成随机盐    │  salt = random(32 bytes)
│     (UUID)      │
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  2.密码+盐组合   │  password + salt = "123456_random_uuid"
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  3.第一次MD5    │  md5_1 = MD5("123456_random_uuid")
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  4.第二次MD5    │  md5_2 = MD5(md5_1)
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  5.SM4加密存储  │  encrypted = SM4_Encrypt(md5_2, key)
└────────┬────────┘
         │
         ▼
┌─────────────────┐
│  数据库存储      │  password = encrypted
│                 │  salt = random_uuid
└─────────────────┘
```

### 2.2 密码验证流程

```java
// 用户登录时密码验证
public boolean verifyPassword(String inputPassword, String storedPassword, String salt) {
    // 1. 使用相同的盐
    String combined = inputPassword + salt;

    // 2. 两次 MD5
    String md5_1 = MD5(combined);
    String md5_2 = MD5(md5_1);

    // 3. SM4 解密存储的密码
    String decrypted = SM4_Decrypt(storedPassword, SM4_KEY);

    // 4. 比对
    return md5_2.equals(decrypted);
}
```

### 2.3 SM4 加密实现

```java
// SM4 加密工具类
public class SM4Util {

    private static final String ALGORITHM = "SM4/ECB/PKCS5Padding";

    /**
     * SM4 加密
     * @param plainText 明文
     * @param key 密钥 (16字节)
     * @return 密文 (十六进制)
     */
    public static String encrypt(String plainText, String key) {
        try {
            Cipher cipher = Cipher.getInstance(ALGORITHM, "BC");
            Key keySpec = new SecretKeySpec(key.getBytes(), "SM4");
            cipher.init(Cipher.ENCRYPT_MODE, keySpec);
            byte[] encrypted = cipher.doFinal(plainText.getBytes(StandardCharsets.UTF_8));
            return Hex.encodeHexString(encrypted);
        } catch (Exception e) {
            throw new CryptoException("SM4 encryption failed", e);
        }
    }

    /**
     * SM4 解密
     * @param cipherText 密文 (十六进制)
     * @param key 密钥 (16字节)
     * @return 明文
     */
    public static String decrypt(String cipherText, String key) {
        try {
            Cipher cipher = Cipher.getInstance(ALGORITHM, "BC");
            Key keySpec = new SecretKeySpec(key.getBytes(), "SM4");
            cipher.init(Cipher.DECRYPT_MODE, keySpec);
            byte[] decrypted = cipher.doFinal(Hex.decodeHex(cipherText.toCharArray()));
            return new String(decrypted, StandardCharsets.UTF_8);
        } catch (Exception e) {
            throw new CryptoException("SM4 decryption failed", e);
        }
    }
}
```

### 3.3 权限校验实现

```java
// 权限注解
@Target({ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
public @interface RequirePermission {
    String[] value();
    Logical logical() default Logical.AND;
}

// 权限校验切面
@Aspect
@Component
public class PermissionAspect {

    @Autowired
    private PermissionService permissionService;

    @Around("@annotation(RequirePermission)")
    public Object checkPermission(ProceedingJoinPoint point) throws Throwable {
        // 1. 获取当前用户
        UserInfo currentUser = getCurrentUser();

        // 2. 获取接口所需权限
        RequirePermission annotation = point.getMethod().getAnnotation(RequirePermission.class);
        String[] requiredPermissions = annotation.value();

        // 3. 校验权限
        boolean hasPermission = permissionService.hasPermissions(
            currentUser.getId(),
            requiredPermissions
        );

        if (!hasPermission) {
            throw new NoPermissionException("无权限访问该接口");
        }

        // 4. 执行目标方法
        return point.proceed();
    }
}

// Controller 使用示例
@RestController
@RequestMapping("/sys/user")
public class SysUserController {

    @RequirePermission("sys:user:add")
    @PostMapping("/add")
    public Result<?> addUser(@RequestBody SysUser user) {
        // 新增用户逻辑
        return Result.ok();
    }

    @RequirePermission("sys:user:edit")
    @PutMapping("/edit")
    public Result<?> editUser(@RequestBody SysUser user) {
        // 编辑用户逻辑
        return Result.ok();
    }
}
```

## 四、数据安全

### 4.1 接口签名验证

```javascript
/**
 * 接口签名算法
 * 1. 将参数按 key 字典序排序
 * 2. 拼接 key=value&key=value 格式
 * 3. 在末尾拼接密钥
 * 4. MD5/SM4 加密生成签名
 */
function generateSign(params, secret) {
    // 1. 排序
    const sortedKeys = Object.keys(params).sort();

    // 2. 拼接
    const signStr = sortedKeys
        .map(key => `${key}=${params[key]}`)
        .join('&') + secret;

    // 3. MD5
    return md5(signStr);
}

// 请求拦截器
axios.interceptors.request.use(config => {
    const timestamp = Date.now();
    const params = { ...config.params, timestamp };

    // 生成签名
    const sign = generateSign(params, SECRET_KEY);

    // 添加签名参数
    config.params = { ...params, sign };

    return config;
});
```

### 4.3 XSS 防护

```java
// XSS 过滤器
@Component
@WebFilter(urlPatterns = "/*")
public class XssFilter implements Filter {

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        HttpServletRequest httpRequest = (HttpServletRequest) request;
        chain.doFilter(new XssRequestWrapper(httpRequest), response);
    }
}

// XSS 请求包装器
public class XssRequestWrapper extends HttpServletRequestWrapper {

    private static final HTMLEscaper HTML_ESCAPER = new HTMLEscaper();

    public XssRequestWrapper(HttpServletRequest request) {
        super(request);
    }

    @Override
    public String getParameter(String name) {
        String value = super.getParameter(name);
        return HTML_ESCAPER.escape(value);
    }

    @Override
    public String[] getParameterValues(String name) {
        String[] values = super.getParameterValues(name);
        if (values == null) return null;
        return Arrays.stream(values).map(HTML_ESCAPER::escape).toArray(String[]::new);
    }
}
```

### 5.2 Spring Security 配置

```java
@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Autowired
    private JwtAuthenticationFilter jwtAuthFilter;

    @Autowired
    private JwtAuthenticationEntryPoint jwtAuthEntryPoint;

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            // 禁用 CSRF
            .csrf().disable()

            // 禁用 Session
            .sessionManagement()
                .sessionCreationPolicy(SessionCreationPolicy.STATELESS)
                .and()

            // 配置路径权限
            .authorizeHttpRequests(auth -> auth
                // 公开路径
                .antMatchers("/sys/login/**", "/captcha/**").permitAll()
                // 其他路径需要认证
                .anyRequest().authenticated()
                .and()

            // 添加 JWT 过滤器
            .addFilterBefore(jwtAuthFilter, UsernamePasswordAuthenticationFilter.class)

            // 异常处理
            .exceptionHandling()
                .authenticationEntryPoint(jwtAuthEntryPoint)
                .and()

            // 禁用缓存
            .headers()
                .cacheControl().disable();

        return http.build();
    }
}
```

### 6.2 操作日志

```java
// 操作日志注解
@Target({ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
public @interface OperLog {
    String value();                    // 操作描述
    OperType operType();              // 操作类型
}

// 操作日志切面
@Aspect
@Component
public class OperLogAspect {

    @Autowired
    private SysLogOperateMapper logMapper;

    @Around("@annotation(OperLog)")
    public Object logOperation(ProceedingJoinPoint point, OperLog operLog) throws Throwable {
        long startTime = System.currentTimeMillis();

        // 执行目标方法
        Object result = point.proceed();

        // 计算耗时
        long costTime = System.currentTimeMillis() - startTime;

        // 获取请求信息
        HttpServletRequest request = getRequest();

        // 保存日志
        SysLogOperate log = new SysLogOperate();
        log.setId(UUID.randomUUID().toString());
        log.setUsername(getCurrentUsername());
        log.setIp(getIpAddress(request));
        log.setMethod(point.getSignature().getName());
        log.setRequestUrl(request.getRequestURI());
        log.setRequestParam(JSON.toJSONString(request.getParameterMap()));
        log.setLogContent(operLog.value());
        log.setOperType(operLog.operType().getCode());
        log.setCostTime(costTime);
        log.setCreateTime(new Date());

        logMapper.insert(log);

        return result;
    }
}

// 使用示例
@RestController
public class SysUserController {

    @OperLog(value = "新增用户", operType = OperType.ADD)
    @PostMapping("/add")
    public Result<?> addUser(@RequestBody SysUser user) {
        userService.save(user);
        return Result.ok();
    }

    @OperLog(value = "删除用户", operType = OperType.DEL)
    @DeleteMapping("/{id}")
    public Result<?> deleteUser(@PathVariable String id) {
        userService.removeById(id);
        return Result.ok();
    }
}
```
