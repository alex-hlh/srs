## 一、测试策略

### 1.1 测试金字塔

```
                              ┌───────────┐

                    ┌────────────┼────────────┐
                   │        集成测试            │    20%
```

```java
@WebMvcTest(SysUserController.class)
class SysUserControllerTest {

    @Autowired
    private MockMvc mockMvc;

    @MockBean
    private SysUserService userService;

    @Autowired
    private ObjectMapper objectMapper;

    @Test
    @DisplayName("用户列表查询 - 成功")
    void testListUsers_Success() throws Exception {
        // Arrange
        List<SysUser> users = Arrays.asList(
            createUser("1", "admin"),
            createUser("2", "user")
        );
        Page<SysUser> page = new Page<>(1, 10, users.size());
        page.setRecords(users);

        when(userService.listUsers(any())).thenReturn(page);

        // Act & Assert
        mockMvc.perform(get("/sys/user/list")
                .param("pageNo", "1")
                .param("pageSize", "10")
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(0))
            .andExpect(jsonPath("$.data.records").isArray())
            .andExpect(jsonPath("$.data.records.length()").value(2))
            .andExpect(jsonPath("$.data.records[0].username").value("admin"));
    }

    @Test
    @DisplayName("用户列表查询 - 分页参数错误")
    void testListUsers_InvalidPage() throws Exception {
        // Act & Assert
        mockMvc.perform(get("/sys/user/list")
                .param("pageNo", "-1")
                .param("pageSize", "10")
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isBadRequest())
            .andExpect(jsonPath("$.code").value(1001));
    }

    @Test
    @DisplayName("新增用户 - 成功")
    void testAddUser_Success() throws Exception {
        // Arrange
        SysUser user = createUser(null, "newuser");
        when(userService.addUser(any())).thenReturn(true);

        // Act & Assert
        mockMvc.perform(post("/sys/user/add")
                .content(objectMapper.writeValueAsString(user))
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(0))
            .andExpect(jsonPath("$.message").value("操作成功"));
    }

    @Test
    @DisplayName("新增用户 - 用户名重复")
    void testAddUser_DuplicateUsername() throws Exception {
        // Arrange
        SysUser user = createUser(null, "existinguser");
        when(userService.addUser(any()))
            .thenThrow(new BusinessException("用户名已存在"));

        // Act & Assert
        mockMvc.perform(post("/sys/user/add")
                .content(objectMapper.writeValueAsString(user))
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(3001))
            .andExpect(jsonPath("$.message").value("用户名已存在"));
    }
}
```

**Service 测试:**

```java
@ExtendWith(MockitoExtension.class)
class SysUserServiceTest {

    @Mock
    private SysUserMapper userMapper;

    @Mock
    private PasswordEncoder passwordEncoder;

    @InjectMocks
    private SysUserServiceImpl userService;

    @Test
    @DisplayName("用户注册 - 密码加密")
    void testRegister_PasswordEncryption() {
        // Arrange
        SysUser user = new SysUser();
        user.setUsername("newuser");
        user.setPassword("plainPassword");
        user.setSalt("randomSalt");

        when(passwordEncoder.encode(anyString())).thenReturn("encryptedPassword");
        when(userMapper.insert(any())).thenReturn(1);

        // Act
        boolean result = userService.register(user);

        // Assert
        assertThat(result).isTrue();
        verify(passwordEncoder).encode("plainPassword");
        verify(userMapper).insert(argThat(u ->
            "encryptedPassword".equals(u.getPassword())
        ));
    }

    @Test
    @DisplayName("用户注册 - 用户名已存在")
    void testRegister_UsernameExists() {
        // Arrange
        SysUser user = new SysUser();
        user.setUsername("existinguser");

        when(userMapper.selectCount(any())).thenReturn(1L);

        // Act & Assert
        assertThatThrownBy(() -> userService.register(user))
            .isInstanceOf(BusinessException.class)
            .hasMessage("用户名已存在");
    }

    @ParameterizedTest
    @CsvSource({
        "admin, true",
        "user, true",
        "test, false",
        "guest, false"
    })
    @DisplayName("用户权限判断")
    void testHasPermission(String username, boolean expected) {
        // Act
        boolean result = userService.isAdmin(username);

        // Assert
        assertThat(result).isEqualTo(expected);
    }
}
```

### 2.3 前端单元测试示例

```javascript
// UserList.spec.js

import { shallowMount } from '@vue/test-utils'
import UserList from '@/views/system/user/UserList.vue'
import { createVue, createRouter } from '../testUtils'

describe('UserList.vue', () => {
  let wrapper
  let mockUserService

  beforeEach(() => {
    mockUserService = {
      list: jest.fn().mockResolvedValue({
        data: {
          records: [
            { id: '1', username: 'admin', realname: '管理员' },
            { id: '2', username: 'user', realname: '用户' }
          ],
          total: 2
        }
      }),
      delete: jest.fn().mockResolvedValue({ code: 0 }),
      freeze: jest.fn().mockResolvedValue({ code: 0 })
    }

    const router = createRouter()
    wrapper = shallowMount(UserList, {
      router,
      mocks: {
        $userService: mockUserService
      }
    })
  })

  afterEach(() => {
    wrapper.destroy()
  })

  describe('数据加载', () => {
    it('应正确加载用户列表', async () => {
      await wrapper.vm.$nextTick()

      expect(mockUserService.list).toHaveBeenCalled()
      expect(wrapper.vm.userList).toHaveLength(2)
      expect(wrapper.findAll('.user-item')).toHaveLength(2)
    })

    it('应正确显示用户名', async () => {
      await wrapper.vm.$nextTick()

      const firstUser = wrapper.find('.user-item:first-child')
      expect(firstUser.text()).toContain('admin')
    })
  })

  describe('搜索功能', () => {
    it('应能按用户名搜索', async () => {
      wrapper.vm.searchForm.username = 'admin'
      wrapper.vm.handleSearch()

      await wrapper.vm.$nextTick()

      expect(mockUserService.list).toHaveBeenCalledWith(
        expect.objectContaining({ username: 'admin' })
      )
    })

    it('搜索后应正确更新列表', async () => {
      mockUserService.list.mockResolvedValueOnce({
        data: { records: [{ id: '1', username: 'admin' }], total: 1 }
      })

      wrapper.vm.searchForm.username = 'admin'
      wrapper.vm.handleSearch()

      await wrapper.vm.$nextTick()

      expect(wrapper.vm.userList).toHaveLength(1)
    })
  })

  describe('删除功能', () => {
    it('应正确删除用户', async () => {
      wrapper.vm.userList = [{ id: '1', username: 'test' }]
      wrapper.vm.currentUser = { id: '2' }

      await wrapper.vm.handleDelete('1')

      expect(mockUserService.delete).toHaveBeenCalledWith('1')
    })

    it('删除后应刷新列表', async () => {
      mockUserService.list.mockResolvedValueOnce({
        data: { records: [], total: 0 }
      })

      wrapper.vm.handleDelete('1')
      await wrapper.vm.$nextTick()

      expect(mockUserService.list).toHaveBeenCalledTimes(2)
    })
  })

  describe('分页功能', () => {
    it('应正确切换页码', async () => {
      wrapper.vm.pagination.current = 2
      wrapper.vm.handlePageChange(2)

      expect(mockUserService.list).toHaveBeenCalledWith(
        expect.objectContaining({ pageNo: 2 })
      )
    })
  })
})
```

### 3.2 API 集成测试

```java
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@AutoConfigureMockMvc
class SysUserApiIntegrationTest {

    @Autowired
    private MockMvc mockMvc;

    @Autowired
    private ObjectMapper objectMapper;

    @MockBean
    private JwtTokenProvider tokenProvider;

    private String adminToken;

    @BeforeEach
    void setUp() {
        adminToken = "Bearer test-token";
        when(tokenProvider.validateToken(adminToken)).thenReturn(true);
        when(tokenProvider.getUserIdFromToken(adminToken)).thenReturn("1");
    }

    @Test
    @DisplayName("创建用户 - 完整流程")
    void testCreateUser_FullFlow() throws Exception {
        // 1. 创建用户
        SysUser newUser = SysUser.builder()
            .username("testuser")
            .realname("测试用户")
            .password("password123")
            .email("test@example.com")
            .build();

        mockMvc.perform(post("/sys/user/add")
                .header("Authorization", adminToken)
                .content(objectMapper.writeValueAsString(newUser))
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(0));

        // 2. 查询验证
        mockMvc.perform(get("/sys/user/list")
                .header("Authorization", adminToken)
                .param("username", "testuser")
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.data.records[0].username").value("testuser"));

        // 3. 编辑用户
        newUser.setRealname("修改后的名称");
        mockMvc.perform(put("/sys/user/edit")
                .header("Authorization", adminToken)
                .content(objectMapper.writeValueAsString(newUser))
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(0));

        // 4. 删除用户
        mockMvc.perform(delete("/sys/user/{id}", newUser.getId())
                .header("Authorization", adminToken)
                .contentType(MediaType.APPLICATION_JSON))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(0));
    }
}
```

```javascript
// tests/e2e/login.spec.js
const { test, expect } = require('@playwright/test')

test.describe('登录功能', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/login')
  })

  test('登录成功', async ({ page }) => {
    // 输入账号密码
    await page.fill('#username', 'admin')
    await page.fill('#password', 'admin123')

    // 点击登录
    await page.click('button[type="submit"]')

    // 验证跳转
    await page.waitForURL('**/dashboard')

    // 验证用户信息
    await expect(page.locator('.user-name')).toContainText('系统管理员')
  })

  test('登录失败 - 密码错误', async ({ page }) => {
    await page.fill('#username', 'admin')
    await page.fill('#password', 'wrongpassword')

    await page.click('button[type="submit"]')

    // 验证错误提示
    await expect(page.locator('.error-message')).toContainText('用户名或密码错误')
  })

  test('登录失败 - 验证码错误', async ({ page }) => {
    await page.fill('#username', 'admin')
    await page.fill('#password', 'admin123')
    await page.fill('#captcha', 'wrongcode')

    await page.click('button[type="submit"]')

    await expect(page.locator('.error-message')).toContainText('验证码错误')
  })
})

test.describe('用户管理', () => {
  let adminPage

  test.beforeEach(async ({ page }) => {
    // 先登录
    adminPage = page
    await adminPage.goto('/login')
    await adminPage.fill('#username', 'admin')
    await adminPage.fill('#password', 'admin123')
    await adminPage.click('button[type="submit"]')
    await adminPage.waitForURL('**/dashboard')

  })

  test('用户列表加载', async ({ page }) => {
    await expect(page.locator('.user-table')).toBeVisible()
    await expect(page.locator('.user-table tr')).toHaveCount(10) // 默认每页10条
  })

  test('搜索用户', async ({ page }) => {
    // 输入搜索条件
    await page.fill('.search-form input[name="username"]', 'admin')
    await page.click('.search-form button[type="submit"]')

    // 等待结果
    await page.waitForSelector('.user-table')

    // 验证结果
    const rows = await page.locator('.user-table tbody tr').all()
    for (const row of rows) {
      const text = await row.textContent()
      expect(text).toContain('admin')
```
